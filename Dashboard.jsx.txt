import { useState, useEffect, useRef } from 'react';
import Sidebar from '../components/Sidebar';
import ChatWindow from '../components/ChatWindow';
import InputBox from '../components/InputBox';
import { useAuth } from '../context/AuthContext';
import axios from 'axios';

const API_URL = 'http://localhost:5000/api'; // Change to Render URL in production

export default function Dashboard() {
  const { user } = useAuth();
  const [messages, setMessages] = useState([]);
  const [isLoading, setIsLoading] = useState(false);
  const [activeChat, setActiveChat] = useState(null);
  const [mode, setMode] = useState('chat'); // chat, notes, summary, questions

  const handleSend = async (text, file = null) => {
    if (!text.trim() && !file) return;

    const newMsg = { role: 'user', content: text };
    setMessages(prev => [...prev, newMsg]);
    setIsLoading(true);

    try {
      let contentToSend = text;

      // If PDF uploaded, process it first
      if (file) {
        const formData = new FormData();
        formData.append('pdf_file', file);
        formData.append('userId', user.uid);
        
        const pdfRes = await axios.post(`${API_URL}/pdf/upload-pdf`, formData);
        contentToSend = `Based on this PDF content: ${pdfRes.data.fullText}\n\nUser question: ${text}`;
      }

      // Send to AI
      const res = await axios.post(`${API_URL}/chat`, {
        userId: user.uid,
        message: contentToSend,
        mode: mode,
        chatId: activeChat || Date.now().toString()
      });

      setMessages(prev => [...prev, { role: 'assistant', content: res.data.response }]);
    } catch (error) {
      console.error(error);
      setMessages(prev => [...prev, { role: 'assistant', content: "Sorry, I couldn't process that." }]);
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="flex h-screen w-full bg-white dark:bg-gray-900">
      <Sidebar onSelectChat={setActiveChat} activeChat={activeChat} />
      
      <div className="flex-1 flex flex-col h-full relative">
        {/* Header */}
        <div className="h-16 border-b dark:border-gray-700 flex items-center justify-between px-6 bg-white dark:bg-gray-800">
          <h1 className="text-xl font-bold bg-gradient-to-r from-blue-600 to-purple-600 bg-clip-text text-transparent">
            AI Study Assistant
          </h1>
          <div className="flex gap-2">
            {['chat', 'notes', 'summary', 'questions'].map((m) => (
              <button
                key={m}
                onClick={() => setMode(m)}
                className={`px-3 py-1 rounded-full text-sm capitalize transition-colors ${
                  mode === m 
                    ? 'bg-blue-100 text-blue-700 dark:bg-blue-900 dark:text-blue-300' 
                    : 'text-gray-500 hover:bg-gray-100 dark:hover:bg-gray-700'
                }`}
              >
                {m}
              </button>
            ))}
          </div>
        </div>

        <ChatWindow messages={messages} isLoading={isLoading} />
        <InputBox onSend={handleSend} isLoading={isLoading} />
      </div>
    </div>
  );
}
